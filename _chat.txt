[2023-04-01, 22:38:17] Nipuna: int lilysHomework(vector<int> arr) {
    int result = INT_MAX;


    vector<int> sorted(arr);
    sort(sorted.begin(), sorted.end(), greater<int>()); //sorting the array in reversed order

    //checking the nuber of times for both ascending and descending
    for (int rev = 0; rev < 2; rev++) {
        int curSwap = 0;
        if (rev) {
            reverse(sorted.begin(), sorted.end());//checking for the unreversed arr
        }
        vector<int> temp(arr);
        // val, pos
        unordered_map<int, int> positions;
        for (int i = 0; i < (int) temp.size(); i++) {
            positions[temp[i]] = i; // keeping tracg of the unsorted indexes
        }


        for (int i = 0; i < (int) temp.size(); i++) {
            if (temp[i] == sorted[i]) {
                continue;
            }
            //if sorted and the normal is not same then sort
            int aindex = temp[i];
            int sindex = sorted[i];
            
            swap(temp[i], temp[positions[sindex]]);
            curSwap++;


            positions[aindex] = positions[sindex];
            positions[sindex] = i;


        }

        result = min(result, curSwap);
    }


    return result;


}

class threadJob implements Runnable{
    private StudyRoomReservationSystem s;
    private int roomNumber;
    private boolean release; // this variable is used to select which operation to be done

    public threadJob(StudyRoomReservationSystem x, int n, boolean m){
        this.s = x;
        this.roomNumber = n;
        this.release = m;
    }
    public void run(){
        if(release){
            StudyRoom x = this.s.rooms.get(roomNumber);
            x.setAvailability(true);
        }else{
            try {
                StudyRoom x = this.s.rooms.get(roomNumber);
                if(!x.getAvailability()){
                    throw new StudyRoomUnavailableException("Study room " + roomNumber + " is not available.");
                }else{
                    x.setAvailability(false);
        
                }
            } catch (StudyRoomUnavailableException e) {
                System.out.println(e.getMessage());
            }
        }
      
    }
}

string ltrim(const string &str) {
    string s(str);

    s.erase(
        s.begin(),
        find_if(s.begin(), s.end(), not1(ptr_fun<int, int>(isspace)))
    );

    return s;
}

string rtrim(const string &str) {
    string s(str);

    s.erase(
        find_if(s.rbegin(), s.rend(), not1(ptr_fun<int, int>(isspace))).base(),
        s.end()
    );

    return s;
